  1. Compute a CaPso mean field model, where the only active transition functions are the reproduction of predators, the death of predators and the death of preys, during 40 seasons using the following parameters:
    • num_iter = 1
    • use_reg = FALSE
    • psi0 = 1
    • phi0 = 0.01
    • ey = 1
    • ry = 1
    • ez = 1
    • rz = 1
rules <- c("rep_preds", "death_preds_adj", "death_preys")
mf1   <- rcapso_compute_mf(rules, num_iter = 40, use_reg = FALSE, psi0 = 1, phi0 = 0.01, ez = 1, rz = 1)
head(mf1)
  1. Repeat the previous step for ez = 3 and ez = 5.
mf3   <- rcapso_compute_mf(rules, num_iter = 40, use_reg = FALSE, psi0 = 1, phi0 = 0.01, ez = 3, rz = 1)
head(mf3)
mf5   <- rcapso_compute_mf(rules, num_iter = 40, use_reg = FALSE, psi0 = 1, phi0 = 0.01, ez = 5, rz = 1)
head(mf5)
  1. Plot the data of each simulation with the corresponding mean field results.
colors <- c("Preys"        = "green",
            "Predators"    = "red",
            "MF Preys"     = "dark green",
            "MF Predators" = "dark red")

lattice_size <- 512 * 512
index_set    <- 1:nrow(sim)

ggplot() +
    geom_line(data = ts1, mapping = aes(x = index_set, y = Preys / lattice_size, color = "Preys")) +
